def second_smallest_eigenvalue_of_normalized_laplacian(G):
    subgraph_laplacian = nx.normalized_laplacian_matrix(G).todense()
    eigenvalues = np.linalg.eigvals(subgraph_laplacian)
    return sorted(eigenvalues)[1]

def subsets(nodes):
